  trustScore: number;        // 0-100
  tier: 'bronze' | 'silver' | 'gold' | 'diamond';

class ScoringEngine {
  /**
   * Calcule le score de confiance d'un client
   */
  async calculateScore(vendorId: string, clientPhone: string): Promise<ClientScore> {
    const client = await storage.getOrCreateClient(vendorId, clientPhone);
    
    // Facteurs positifs
    let score = 50; // Base
    score += Math.min(client.successfulPayments * 5, 25);   // Max +25 pour paiements
    score += client.avgPaymentTimeSeconds < 120 ? 10 : 0;   // Rapide = +10
    score += client.totalSpent > 100000 ? 10 : 0;           // Gros client = +10
    
    // Facteurs nÃ©gatifs
    score -= client.expiredReservations * 10;               // -10 par expiration
    
    // DÃ©tection patterns frauduleux
    const riskFlags: string[] = [];
    if (client.expiredReservations > 3 && client.successfulPayments < 2) {
      riskFlags.push('faux_acheteur_potentiel');
      score -= 20;
    }
    
    score = Math.max(0, Math.min(100, score));
    
    return {
      trustScore: score,
      tier: this.getTier(score, client.totalOrders),
      riskFlags,
      recommendations: this.getRecommendations(score, riskFlags)
    };
  }
  
  private getTier(score: number, orders: number): Tier {
    if (orders >= 10 && score >= 80) return 'diamond';
    if (orders >= 6 && score >= 70) return 'gold';
    if (orders >= 3 && score >= 60) return 'silver';
    return 'bronze';
  }
  
  private getRecommendations(score: number, flags: string[]) {
    if (score < 30 || flags.includes('faux_acheteur_potentiel')) {
      return {
        reservationMinutes: 5,
        priorityLevel: 'normal' as const,
        requirePrePayment: true
      };
    }
    if (score >= 80) {
      return {
        reservationMinutes: 15,
        priorityLevel: 'vip' as const,
        requirePrePayment: false
      };
    }
    return {
      reservationMinutes: 10,
      priorityLevel: 'normal' as const,
      requirePrePayment: false
    };
  }
